(function () { const menuSections = Array.from(document.querySelectorAll("[data-menu-section]")); function updateMark(section) { const mark = section.querySelector(".menu-toggle-mark"); if (!mark) { return; } mark.textContent = section.classList.contains("is-open") ? "-" : "+"; } function closeOthers(currentSection) { menuSections.forEach((section) => { if (section === currentSection) { return; } section.classList.remove("is-open"); updateMark(section); }); } const openSections = menuSections.filter((section) => section.classList.contains("is-open")); if (openSections.length > 1) { const preferredOpenSection = openSections.find((section) => section.querySelector(".menu-item-link.active, .friend-item.active")) || openSections[0]; closeOthers(preferredOpenSection); } menuSections.forEach((section) => updateMark(section)); menuSections.forEach((section) => { const toggle = section.querySelector("[data-menu-toggle]"); if (!toggle) { return; } toggle.addEventListener("click", () => { closeOthers(section); section.classList.add("is-open"); updateMark(section); }); }); })(); (function () { const form = document.querySelector(".user-theme-form"); const toggle = form ? form.querySelector(".js-sidebar-theme-toggle") : null; const valueInput = form ? form.querySelector(".js-sidebar-theme-value") : null; if (!form || !toggle || !valueInput) { return; } toggle.addEventListener("change", function () { valueInput.value = toggle.checked ? "light" : "dark"; form.submit(); }); })();
